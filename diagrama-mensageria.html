4. Como o código consome

API genérica (recomendada)

from lavup.services.mensagens import MensagemService

# Reagendamento confirmado
MensagemService.enviar(
    'agendamento_confirmado',
    cliente.whatsapp,
    data='09/05',
    horario='14:00',
    servicos='Lavagem + Cera',
)

Wrapper compatível (legado)

from lavup.services.mensagens import MensagemService

# Mantido para auth/views existentes
MensagemService.enviar_codigo_verificacao(
    usuario.whatsapp,
    codigo.codigo,
)
